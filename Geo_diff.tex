\begin{Prop}
Soient $X,Y$ deux variétés de même dimension, $X$ est supposée compacte. Soient $f : X \to Y$ une application lisse et $y$ une valeur régulière de $f$. Alors,
\begin{itemize}
	\item $f^{-1}(y)$ est fini
	\item Il existe un voisinage ouvert $V$ de $y$ tel que :
		\[ \forall z \in V, | f^{-1}(z)|=|f^{-1}(y)|
	\]
\end{itemize}
\end{Prop}
\begin{proof}
Lafontaine p 58
\end{proof}
\begin{Cor}
$f$ est un revêtement sur l'ensemble des valeurs régulières.
\end{Cor}

\begin{Thm}
Soit $X$ une variété compacte connexe orientée de dimension $n$. Une forme différentielle de degré $n$ est exacte si, et seulement si, elle est d'intégrale nulle
\end{Thm}
