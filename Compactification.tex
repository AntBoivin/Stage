\section{Action compactifiable}

\begin{Def}
Soit $X$ un espace analytique complexe compact. Soit $H$ un groupe de Lie complexe connexe agissant sur $X$. On dit $X$ agit de façon compactifiable s'il existe un espace analytique complexe compact $Y$, une application holomorphe injective $i : H \to Y$ telle que $i(H)$ est un ouvert de Zariski de $Y$ et une application méromorphe $m : Y \times X \dashrightarrow X$ définit sur $i(H) \times X$ par : 
	\[ m(i(h),x)=h \cdot x
\]
\end{Def}
Lorsque l'on a une action compactifible de $H$ sur $X$, on peut munir $H$ d'une topologie, dite de Zariski, définie par le fait que cses fermés sont de la forme $i^{-1}(V)$ où $V$ est un sous-espace analytique fermé de $Y$. On appelera $\Zcal$-ouverts (resp. $\Zcal$-fermés) les ouverts (resp. les fermés) de $H$ pour cette topologie (que l'on appelera $\Zcal$-topologie sur $H$). \\
Cette topologie est, en général, plus grossière que la topologie de Zariski usuelle de $H$ : 
Prenons $X=\P^n$ et $H=Aut_0(X)=PGL_{n+1}(\C)$ (voir \cite{Aut(Pn)} que l'on peut compactifier avec $Y=\P^{(n+1)^2-1}=\P(\Mcal_{n+1}(\C))$ et l'application méromorphe : 
	\[ m : ([(c_{ij})],[(x_j)] \in Y \times \P^n \mapsto \left[\sum_j c_{ij} x_j\right] \in \P^n
\]
Les sous-espaces analytiques de $\P^N$ sont des sous-ensembles algébriques, par le théorème de Chow. On en déduit que les $\Zcal$-ouverts de $H$ sont les sous-ensembles algébriques de celui-ci.

\begin{Prop} % On suppose Y normal
Soit $H$ un groupe de Lie complexe connexe agissant effectivement sur un espace analytique complexe compact $X$. L'action est compactifiable si, et seulement si, le graphe des éléments de $H$ forme un ouvert de Zariski d'un sous-espace analytique compact $\Cscr(X \times X)$.
\end{Prop}
\begin{proof}
$\Leftarrow$ : On prend l'adhérence de $H$ dans $\Cscr(X \times X)$. \\
$\Rightarrow$ : Supposons $H$ compactifiable. Il existe donc un compact $Y$ tel que $H$ s'injecte dans $Y$ et une application méromorphe $m : Y \times X \dashrightarrow X$. De ce fait, par définition, il existe $\Gamma \subset Y \times X \times X$ tel que la projection $\Gamma \to Y \times X$ soit une modification (propre) et $\Gamma \to X$ est holomorphe. Comme $X$ est compact alors la projection $\Gamma \to Y$ est propre (et surjective). On peut donc appliquer la proposition \ref{merom} et obtenir une application méromorphe $Y \dashrightarrow \Cscr(X \times X)$ qui étend l'application $H \to \Cscr(X \times X)$ définie par $h \mapsto \Gamma \cap \{h\}\times X \times X$. \\
Il existe donc un espace analytique complexe $Y'$, une modification (propre) $\tau : Y' \to Y$ et une application holomorphe $f : Y' \to \Cscr(X \times X)$. Comme $\tau$ est propre et que $Y$ est compact alors $Y'=\tau^{-1}(Y)$ est compact et donc $f(Y')$ est le compact désiré.

\end{proof}
Par cette proposition, la $\Zcal$-topologie de $H$ est donnée par celle induite par $\Cscr(X \times X)$ et ne dépend pas de la compactification $Y$.

\begin{Prop}
Soit $H$ agissant de façon compactifiable sur $X$. Un sous-groupe $S \subset H$ agit de façon compactifiable sur $X$ si, et seulement si, $S$ est $\Zcal$-fermé.

\end{Prop}
\begin{proof}
Notons $Y$ la compactification de $H$. \\
$\Leftarrow$ : Si $S$ est $\Zcal$-fermé alors c'est l'intersection d'un fermé de $\Cscr(X \times X)$ et de $H$. C'est donc un ouvert de Zariski de l'intersection d'un fermé de $\Cscr(X \times X)$ et de $Y$ qui est compact
$\Rightarrow$ : Soit $Y'$ une compactification de $S$. Quitte à l'intersecter avec $H$, on peut supposer $Y' \subset H$. Comme les opérations de groupe de $H$ sont holomorphes alors $H$ muni de sa topologie de Zariski est un groupe topologique. \\
Par hypothèse, $S$ est $\Zcal$-ouvert dans son $\Zcal$-adhérence alors $S$ est localement fermé et donc, par la proposition \ref{localement_ferme}, est ($\Zcal$-)fermé dans $H$
\end{proof}
\begin{Thm}\label{Aut_0_comp}
Si $X$ est une variété de Kähler compacte alors $Aut_0(X)$ agit de façon compactifiable sur $X$

\end{Thm}
\begin{proof}
Par le théorème \ref{biholo}, $Aut_0(X)$ est un ouvert de Zariski de $\Cscr_\Delta(X \times X)$ la composante irréductible du graphe de l'identité. Par le théorème \ref{compact}, $\Cscr_\Delta(X \times X)$ est compact, ce qui permet de conclure.
\end{proof}
\begin{Def}
Soit $H$ agissant sur $X$. On définit les deux ensembles :
	\[H_{\{Z\}}:=\{ h \in H \mid h \cdot Z=Z\}
\]
\[H_{Z}:=\{ h \in H \mid \forall z \in Z, h \cdot z=z\}
\]

\end{Def}
On peut remarquer que $H_{\{Z\}}$ est un groupe.
\begin{Prop} \label{fix_topol}
Si $H$ agit de façon compactifiable sur $X$ alors $H_Z$ est un sous-groupe $\Zcal$-fermé de $H$. Si $Z$ est Zariski-fermé alors $H_{\{Z\}}$ est $\Zcal$-fermé.

\end{Prop}
\begin{proof}
Soit $Y$ la compactification de $H$. \\
On peut commencer par remarquer que :
	\[H_{\{Z\}}=\bigcap_{z \in Z} \{h \in H \mid hz \in Z \}
\]
Puis, en notant $\widetilde{m}$ l'application holomorphe du graphe de $m$, $\Gamma$ dans $X$ et $\tau$ la modification propre $\Gamma \to Y \times X$, on obtient que l'ensemble $\tau(\widetilde{m}(\cdot,z)^{-1}(Z))$ est un sous-espace analytique de $Y$ (si $Z$ en est un lui-même) et que 
	\[\{h \in H \mid hz \in Z \}=i^{-1}\left((\widetilde{m}(\cdot,z))^{-1}(Z)\right)
\]
C'est donc un $\Zcal$-fermé de $H$. On en déduit que $H_{\{Z\}}$ est aussi un $\Zcal$-fermé de $H$. \\
La première partie se montre en remarquant que 
	\[ H_Z=\bigcap_{z \in Z} H_\{z\}
\]
Tous les ensembles de l'intersection sont $\Zcal$-fermés et donc $H_Z$ l'est aussi.
\end{proof}
\begin{Prop} \label{centr_norma}
Soit $H$ qui agit de façon compactifiable sur $X$. Soit $M$ un sous-ensemble de $H$. Alors, le centralisateur de $M$, défini par :
	\[Z(M):=\{ h \in H \mid \forall m \in M, hm=mh \}
\]
est $\Zcal$-fermé. \\
Si $M$ est un fermé de Zariski alors son normalisateur, défini par :
\[N(M):=\{ h \in H \mid hMh^{-1}=M \}
\]
est $\Zcal$-fermé. 
\end{Prop}
\begin{proof}
Soit $Y \subset \Cscr(X \times X)$ la compactification de $H$. L'action de $H$ sur $X \times X$ définie par $h \cdot(a,b)=(h\cdot a,h \cdot b)$ induit une action de $H$ sur $\Cscr(X \times X)$ qui induit la conjugaison $k \mapsto hkh^{-1}$ sur $H$ (car l'identification de $H$ avec un sous-ensemble de $\Cscr(X \times H)$ se fait avec les graphes de l'action et donc si $k =\Gamma_f$ alors :
	\[ h \cdot k=\{ (ha,hf(a)) \in X \times X \mid a \in X\}=\{ (a,hf(h^{-1}a)) \in X \times X \mid a \in X\}
\]
Et donc, $h \cdot k=\Gamma_{hfh^{-1}}$). \\
 Par continuité, l'action de $H$ préserve $Y$, son adhérence dans $\Cscr(X \times X)$. \\
D'autre part, l'action de $H$ sur $X$ est compactifiable i.e. il existe une application méromorphe $m' : Y \times X \times X \dashrightarrow X \times X$ définie par $m'(y,a,b):=(m(y,a),m(y,b))$ qui prolonge l'action de $H$ sur $X \times X$. Notons par $\tau : Y' \to Y \times X \times X$ la modification associée et $\widetilde{m'} : Y'\to X$ la fonction holomorphe prolongeant $m'$. On peut donc passer à l'image directe et obtenir des applications : \\
\xymatrix{
    \Cscr(Y \times X \times X)    && \Cscr(X \times X)  \\
    &\Cscr(Y') \ar[ul]\ar[ur] & 
  }\\
On peut ensuite injecter $Y \times \Cscr(X \times X)$ dans $\Cscr(Y \times X \times X)$ grâce à l'application : 
	\[ i : (y,\sum n_i Z_i) \in Y \times \Cscr(X \times X) \mapsto \sum n_i \{y\} \times Z_i
\]
L'application $\tau_*^{-1}(Im(i)) \subset \Cscr(Y') \to Y \times \Cscr(X \times X)$ est une modification propre (grâce au fait que $\tau$ l'est et $Y'$ est compact). Ce qui montre que l'appplication obtenue $Y \times \Cscr(X \times X) \to \Cscr(X \times X)$ est méromorphe et donc que $H$ agit de manière compactifiable sur $\Cscr(X \times X)$. On conclut en remarquant que, en notant $h \cdot x=hxh^{-1}$ l'action de $h \in H$ sur $x \in H$, 
	\[Z(M)=\{ h \in H \mid \forall m \in M, hm=mh\}=\{ h \in H \mid \forall m \in M, h\cdot m=m\}=H_M
\]
et 
\[N(M)=\{ h \in H \mid  hMh^{-1}=M\}=\{ h \in H \mid h\cdot M=M\}=H_{\{M\}}
\]
et en utilisant la proposition \ref{fix_topol}.
\end{proof}

\begin{Lemme} \label{centralisa}
Soit $S$ un sous-groupe abélien d'un groupe $G$. Alors, $Z(Z(S))$ est un sous-groupe abélien de $G$ qui contient $S$.

\end{Lemme}
\begin{proof}
$S$ est contenu dans $Z(Z(S))$ car, par définition de $Z(S)$, tous ces éléments commutent avec ceux de $S$. \\
Comme $S$ est abélien alors $S \subset Z(S)=\{ h \in H \mid \forall s \in S, hs=sh \}$. On peut ensuite remarquer le fait général que si $X \subset Y$ alors $Z(Y) \subset Z(S)$ (vu que l'on impose moins de conditions sur les éléments). Ainsi, $Z(Z(S)) \subset Z(S)$. \\
On en déduit que si $x,y \in Z(Z(S)) \subset Z(S)$ alors $xy=yx$ (les éléments de $Z(Z(S))$ commutent avec ceux de $Z(S)$)
\end{proof}
\begin{Cor}
Soit $H$ qui agit de façon compactifiable sur $X$ et $S$ un sous-groupe abélien de $H$. Alors, sa $\Zcal$-adhérence $\overline{S}$ est abélienne.

\end{Cor}
\begin{proof}
Par la proposition \ref{centr_norma} et le lemme \ref{centralisa}, $Z(Z(S))$ est un sous-groupe $\Zcal$-fermé abélien contenant $S$. L'adhérence de $S$ est donc contenue dedans et est donc abélienne.
\end{proof}

\begin{Prop} \label{comp_orb_adh}
Soit $H$ qui agit de façon compactifiable sur $X$. Un sous-groupe $S$ de $H$ est $\Zcal$-fermé si, et seulement si, pour toute ensemble fini de points $x_1,\ldots,x_n$ de $X$ , la $S$-orbite $(sx_1,\ldots,sx_n)$ est ouverte dans son adhérence de Zariski.

\end{Prop}

\begin{proof}
$\Rightarrow$ :Si $S$ est $\Zcal$-fermé dans $H$ alors $S$ agit de façon compactifiable. Notons $Y$ sa compactification et $\Gamma$ le graphe de l'application méromorphe $m : Y \times X \dashrightarrow X$ et $\pi_3 : \Gamma \to X$ la projection canonique. 
Montrons l'implication pour $n=1$. Cela suffit car on peut ensuite remarquer que l'action de $S$ sur $X^n$ est compactifiable avec l'application méromorphe :
	\[ m(y,x_1,\ldots,x_n)=(m(y,x_1),\ldots,m(y,x_n))
\]
$\Gamma \cap S \times X \times X$ est un ouvert de Zariski de $\Gamma$ car $S$ est un ouvert de Zariski de $Y$. Ainsi, l'espace analytique $\pi_3(\pi_3^{-1}(x))$ contient un ouvert de Zariski $U$ contenu dans $S \cdot x$. Donc $S \cdot x=\bigcup_{h \in H} h \cdot U$ est un ouvert de Zariski de son adhérence. \\
$\Leftarrow$ : Soient $x_1, \ldots,x_n \in X$ tels que l'application $\lambda : H \to X^n$ définie par :
	\[\lambda(h):=(h \cdot x_1,\ldots,h \cdot x_n)
\]
soit injective (on peut toujours le faire car la suite d'intersection $\left( \bigcap_{i=1}^n {Aut_0}_{x_i} \right)$ finira toujours par être réduit )à l'identité en un nombre fini d'étapes). Par hypothèse et la proposition \ref{localement_ferme}, $\lambda(S)$ est Zariski-fermé dans $\lambda(H)$ et donc $S$ est $\Zcal$-fermé dans $H$ (par continuité), $S$ est donc compactifiable.
\end{proof}
Le début de la preuve (i.e l'implication) s'adapte si on remplace les points $x_1,\ldots, x_n$ par un fermé de Zariski de $X$. \\

On ne peut pas étendre ce résultat au produit cartésien de deux espaces analytiques distincts, même de Kähler : 
\begin{Ex}[Serre]
Soit $S=\C^* \times \C^*$. \\
Une première compactification de $S$ est $X_1=\P^1 \times \P^1$.  Construisons-en une deuxième : \\
On commence par remarquer\footnote{cela découle de l'isomorphisme $\C/\Z \simeq \C^*$ obtenue par passage au quotient de $\exp(2i \pi \cdot)$} que $S\simeq \C^2/\Z^2$. Soit $f : \C^2 \to \C$ l'application linéaire donnée par : $f((1,0))=1$ et $f((0,1))=i$. De cette façon, $f(\Z^2)=\Z \oplus \Z i$. Par construction, l'application $\pi \circ f : \C^2 \to \C/(\Z \oplus \Z i)=:E$ passe au quotient et on obtient un morphisme de groupes holomorphe $S \to E$ de noyau $\{(x,ix)\mid x \in \C\} \simeq \C$. Toutes les autres fibres sont isomorphes à $\C$ ;  peut donc les compactifier en $\P^1$. Le fibré $X_2$ à fibre $\P^1$ sur $E$ ainsi obtenu est un compact contenant $S$. \\
$S$ agit de façon compactifiable sur $X_1$ et $X_2$. \\
Soit $x_i \in X_i$, $i=1,2$ dont leur orbite par $S$ est Zariski-dense dans $X_i$. Supposons que $S\cdot( X_1,x_2)$ est un ouvert de Zariski de son adhérence dans $X_1 \times X_2$. On a alors le diagramme : 
\xymatrix{
    S\cdot x_1 \subset X_1 \ar@{-->}[rr]   && S\cdot x_2 \subset X_2 \\
    &S\cdot(x_1,x_2)\ar[ul]\ar[ur] & 
  }\\
On obtient donc une application biméromorphe de $X_1$ et $X_2$, ce qui est absurde\footnote{Pour voir que $X_1$ et $X_2$ ne sont pas biméromorphes, on peut regarder l'irrégularité de ces deux surfaces i.e. l'entier $q:=h^{0,1}$ qui est un invariant par biméromorphismes (voir \ref{\ccs} Chapitre 3 corollaire 6.4 p 107). \\
Par la proposition 4.1 du chapitre 5 du même livre, $X_2$ est biméromorphe avec $\P^1 \times E$. Ensuite, par le théorème de Hodge, $q(X_i)=b_1(X_i)/2$ (car $\P^1$ et $E$ sont de Kähler, où $b_i$ est le $i$ème nombre de Betti). \\
Par la formule de Künneth, 
	\[b_1(X_1)=b_0(\P^1)b_1(\P^1)+b_1(\P^1)b_0(\P^1)=0
\]
car $\P^1$ est connexe et simplement connexe. \\
Et, 
\[b_1(X_2)=b_0(\P^1)b_1(C)+b_1(\P^1)b_0(C)=2
\]
car $b_1(C)=2h^{1,0}=2\times g(C)=2$. \\
On en déduit que $q(X_1)=0 \neq 1 =q(X_2)$ et donc que $X_1$ et $X_2$ ne sont pas biméromorphes.}

\end{Ex}

Soit $f : X \to Y$ une fonction holomorphe équivariante pour une action de $H$ sur $X$ et $Y$. Supposons que $H$ agit de façon compactifiable sur $X$ et que $Y$ est de Kähler. \\
L'équivariance de $f$ fait que l'automorphisme induit par $h \in H$ conserve $f(X)$ i.e. on a une morphisme de groupes holomorphe $H \to Aut_0(Y)_{\{f(X)\}}$ que l'on peut ensuite composer avec la projection canonique 
	\[Aut_0(Y)_{\{f(X)\}} \to Aut_0(Y)_{\{f(X)\}} /Aut_0(Y)_{f(X)}=:G
\]
 pour obtenir un morphisme $\lambda : H \to G$.

\begin{Prop} \label{image_noyau_Zar}
Le noyau de $\lambda$  est $\Zcal$-fermé et l'image de $\lambda$ est un fermé de $G$ pour la ($\Zcal$-)topologie quotient.
\end{Prop}
\begin{proof}
  Montrons que $\Ker(\lambda)$ est $\Zcal$-fermé :
  Pour cela, il suffit de montrer l'égalité :
   \[
    \Ker(\lambda)=\bigcap_{y \in f(X)}H_{\{f^{-1}(y)\}}
  \]
  et d'utiliser la proposition \ref{fix_topol}. \\
  On a, par définition,
  \[
    \Ker(\lambda)=\{ h \in H \mid \forall y\in f(X), h \cdot y=y \}
  \]
  Soit $h \in \Ker(\lambda)$. Soient $y \in f(X)$ et $x \in f^{-1}(y)$. Alors,
  \[
    f(h \cdot x)=h \cdot f(x)=h \cdot y=y
  \]
  i.e $h \cdot x \in f^{-1}(y)$. \\
  Le même raisonnement pour $h^{-1}$ nous montre que $h \in H_{\{f^{-1}(y)\}}$.    \\
  Cette égalité étant vraie pour tout $y \in Y$, on obtient l'inclusion :  $\Ker(\lambda) \subset \bigcap_{y \in f(X)}H_{\{f^{-1}(y)\}}$. \\
  Réciproquement, si $h \in \bigcap_{y \in f(X)}H_{\{f^{-1}(y)\}}$, alors pour tout $y \in f(X)$ et $x \in X$ tel que $y=f(x)$, \\
  \[
    h \cdot y=h\cdot f(x)=f(h \cdot x)=y
  \]
  car $x \in f^{-1}(y)$. Cela montre que $ h \in Ker(\lambda)$, ce qui montre l'égalité voulue. \\
  Montrons maintenant que $Im(\lambda)$ est un fermé de Zariski. \\
  On peut commencer par remarquer que $G$ agit de façon compactifiable sur $f(X)$. En effet, par \ref{Aut_0_comp}, $Aut_0(Y)$ agit de façon compactifiable sur $Y$. Comme $Aut_0(Y)_{\{f(X)\}}$ est un $\Zcal$-fermé (par \ref{fix_topol} et le fait que $f(X)$ est un fermé de Zariski car $f$ est propre) alors $Aut_0(Y)_{\{f(X)\}}$ agit de façon compactifiable sur $Y$ et en particulier sur $f(X)$. Comme $G$ est le quotient de $Aut_0(Y)_{\{f(X)\}}$ par le sous-groupe d'ineffectivité, l'action passe au quotient et on obtient une action de $G$ sur $f(X)$ compactifiable. \\
  D'après \ref{comp_orb_adh}, on peut choisir $y_1,\ldots,y_n \in f(X)$ de tel sorte que l'application $\varphi : G \to Y^n$ définie , pour tout $h \in H$, par :
  \[
    \varphi(g)=(g\cdot y_1,\ldots,g \cdot y_n)
  \]
  permette d'identifier $G$ avec un ouvert de Zariski de $\overline{\varphi(G)}$. \\
  Soit $x_i$ tel que $y_i=f(x_i)$. Par la même proposition, on obtient une application 
		\[\psi : h\in H \mapsto (h\cdot x_1,\ldots,h\cdot x_n) \in X^n
	\]
	 qui identifie $H$ avec un ouvert de Zariski de $\overline{\psi(H)}$. \\
  Comme $f$ est propre alors $f \times \ldots \times f(\overline{\psi(H)})$ est un fermé de Zariski et par \ref{image_constructible}, $f \times \ldots \times f(\psi(H))$ est un ensemble constructible de $f \times \ldots \times f( \overline{\psi(H)})$. On peut ensuite remarquer que le diagramme: 
 
	 \[\xymatrix{
			 H \ar[r]^\varphi \ar[d]_\lambda  & X^n \ar[d]^{f \times \cdot \times f} \\
			G \ar[r]^\psi & Y^n
		}
\]
  est commutatif. En effet, pour tout $h \in H$,
  \[
    (f \times \cdot \times f) \circ \psi(h)= (f \times \cdot \times f)(h\cdot x_1,\ldots,h \cdot x_n)=(f(h \cdot x_1),\cdots, f(h \cdots x_n))=(h\cdot y_1,\ldots,h \cdot y_n)
  \]
  et
  \[
    \varphi(\lambda(h))=\varphi(h : x \mapsto h \cdot x)=(h\cdot y_1,\ldots,h \cdot y_n)
  \]
  Par conséquent, $f \times \ldots \times f(\psi(H))$ est inclus dans $ \varphi(G) $. Ce qui veut dire que $f \times \ldots \times f(\psi(H))$ est une partie constructible de $\varphi(G)$. L'identification donnée par $\varphi$ munit $\varphi(G)$ d'une structure de groupe topologique pour laquelle $f \times \ldots \times f(\psi(H))$ est un sous-groupe. Par \ref{image_constructible}, $f \times \ldots \times f(\psi(H))$ est un sous-groupe fermé (pour la topologie de Zariski) et comme $\lambda(H)=\varphi^{-1}(f \times \ldots \times f(\psi(H))$ alors $\lambda(H)$ est un fermé de Zariski.

\end{proof}


\begin{Ex}
  Soient $f$, $X$ et $Y$ comme dans \ref{image_noyau_Zar}, $H$ un sous-groupe Zariski-fermé de $Aut_0(X)$ et soit $\lambda : H \to Aut_0(Y)$ définie par $h \mapsto (y \mapsto g(y))$. Alors $\lambda(H)$ n'est pas nécessairement $\Zcal$-fermé dans $Aut_0(Y)$ : \\
  Soient $X=\P^1$, $Y=X \times \P^1$ et $H=\C$. $H$ agit sur $X$ par :
  \[t \cdot [x:y] = \begin{pmatrix} 1 & t \\ 0 & 1 \end{pmatrix}[x:y]=[x+ty : y]\]
  et sur la deuxième composante de $Y$ par :
    \[t \cdot [x:y] = \begin{pmatrix} 1 & 0 \\ 0 & e^t \end{pmatrix}[x:y]=[x :e^t y]\]
    On peut commencer par remarquer que $H$ agit de façon compactifiable sur $X$ avec l'injection $\C \to \P^1$ et l'action méromorphe donnée pour $x \in \P^1$ par :
    \[
      [1:0] \cdot x=[1:0]
    \]
    car, si $x_2 \neq 0$, $\lim \limits_{t \to \infty}x_1+tx_2=\infty$ et si $x_2=0$ alors pour tout $t \in \C$, $t \cdot [x_1,0]=[1:0]$.
    On peut aussi remarquer que $Y$ est de Kähler comme produit de variétés de Kähler et que $Aut_0(Y)=Aut_0(\P^1) \times Aut_0(\P^1)=PGL_2(\C)^2$ (voir \ref{Aut(Pn)} et \ref{prod_aut} ). \\
    On obtient donc que :
    \[\lambda(H)=\left\{\left(\begin{pmatrix} 1 & t \\ 0 & 1 \end{pmatrix},\begin{pmatrix}1 & 0 \\ 0 &e^t \end{pmatrix}\right) \mid t \in \R \right\}
    \]
    qui n'est pas $\Zcal$-fermé car la $\Zcal$-topologie sur $PGL_n(\C)$ est la topologie de Zariski algébrique (voir le début de la section) et la présence de l'exponentielle empêche que cela soit algébrique.
\end{Ex}
