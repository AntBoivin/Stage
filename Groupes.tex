\section{Groupes}
Le but de cette section est de redonner quelques résultats sur les groupes topologiques et sur les groupes de Lie complexes afin d'étudier le groupe d'automorphismes d'un espace analytique dans la section \ref{partie_autom} et les actions compactifiables dans la partie \ref{compactifiable} 


\subsection{ Groupes topologiques}

\begin{Def} \index{groupe topologique}
Un groupe topologique est un groupe $(G,*)$ dont l'ensemble sous-jacent est muni d'une topologie compatible avec la loi du groupe i.e. les applications $(x,y)  \in G \times G \mapsto x*y \in G$ et $x \in G \mapsto x^{-1} \in G$ sont continues. Autrement dit, un groupe topologique est un groupe dans la catégorie des espaces topologiques.

\end{Def}
On dira qu'un groupe topologique est connexe, séparé, compact,... si son espace topologique sous-jacent l'est.
\begin{Prop} \label{identity_component}
La composante connexe du neutre $G_0$ est un sous-groupe fermé distingué de $G$
\end{Prop}
\begin{proof}
Le sous-ensemble $G_0$ est un sous-groupe de $G$. Par définition, le neutre est dans $G_0$, il nous suffit de montrer que : 
	\[\forall x,y \in G_0,xy^{-1} \in G_0
	\]
	 \\
Pour cela, on considère l'application $\mu : (x,y) \in G_0 \times G_0 \mapsto xy^{-1} \in G$ qui est continue et comme $G_0 \times G_0$ est connexe alors son image aussi. Comme, de plus, le neutre est dans cette image alors, $\mu$ est à valeurs dans $G_0$, ce qui montre le résultat \\
On montre de la même façon que $G_0$ est distingué en considérant,pour tout $x \in G$, l'application $h \in G_0 \mapsto xhx^{-1}$.\\
Montrons maintenant que $G_0$ est fermé : \\
L'adhérence $\overline{G_0}$ de $G_0$ est connexe (car $G_0$ l'est). Par conséquent, $\overline{G_0}\subset G_0$ et donc $\overline{G_0}=G_0$

\end{proof}

$G_0$ n'est pas nécessairement ouvert : si $G=(\Q,+)$ est le groupe des rationnels alors $G_0$ est un singleton et n'est donc pas ouvert.



\begin{Prop}\label{engendr}
Soit $G$ un groupe topologique connexe et $V$ un voisinage ouvert de l'identité. Alors $V$ engendre $G$. 

\end{Prop}

\begin{proof}
Soit $H$ le sous-groupe engendré par $V$. Montrons que $G=H$ \\
Comme $H=\bigcup_{h \in H} hV$ et que les $hV$ sont ouverts, car homéomorphes à $V$, alors $H$ est ouvert. \\
Le groupe $G$ est la réunion disjointe de classes d'équivalences $g_iH$, toutes homéo\-morphes à $H$. De ce fait, $H$ est le complémentaire de l'union des autres classes d'équivalence et est donc fermé. \\
Comme $G$ est connexe alors $G=H$.

\end{proof}

\begin{Lemme}\label{adh_groupe}
Soit $G$ un groupe topologique et $H$ un sous-groupe de $G$. Alors $\overline{H}$ est un sous-groupe de $G$.

\end{Lemme}
\begin{proof}
Pour montrer que $\overline{H}$ est un sous-groupe de $G$, il suffit de montrer que $\overline{H}$ est stable pour la loi de groupe et par passage à l'inverse. \\
Soit $a \in \overline{H}$. Soit $U$ un voisinage ouvert de $a^{-1}$ dans $G$. \\
Comme le passage à l'inverse est un homéomorphisme (et est donc ouvert) alors $V=\{ v \in G | v^{-1} \in U\}$ est un voisinage de $a$ dans $G$. \\
Comme $a \in \overline{H}$ alors $V \cap H \neq \emptyset$ et comme $H$ est stable par passage à l'inverse alors $U \cap H \neq \emptyset$. On en déduit que $a^{-1} \in \overline{H}$. \\
Soit $a,b \in \overline{H}$. Soit $W$ un voisinage ouvert de $ab$. \\
Alors, $m^{-1}(ab)$ est un voisinage ouvert de $(a,b)$ dans $G \times G$ ($m : G \times G$ est la loi du groupe). Il existe donc des ouverts $U$ et $V$ de $G$ tels que $U \times V \subset m^{-1}(ab)$. \\
Comme $a,b \in \overline{H}$ alors $H \cap U\neq \emptyset$ et $H \cap V\neq \emptyset$. Ainsi, $m(U \times V) \cap H \neq \emptyset$ et donc $W \cap H \neq \emptyset$. On en déduit que $ab \in \overline{H}$. 
\end{proof}
\begin{Def}
Soit $X$ un espace topologique. Un sous-ensemble de $X$ est dit localement fermé s'il est ouvert dans son adhérence ou encore que c'est l'intersection d'un ouvert et d'un fermé de $X$.
\end{Def}

\begin{Def}
Soit $X$ un espace topologique \footnote{On doit imposer que cela soit un espace noethérien mais dans notre cadre, cela n'est pas restrictif. On supposera dans la suite que nos espaces topologiques sont noethériens}. Un sous-ensemble de $X$ est dit constructible si c'est l'union finie de sous-espaces localement fermés.
\end{Def}
\begin{Lemme}
Une partie constructible $Y$ d'un espace topologique contient un ouvert dense dans $\overline{Y}$. 

\end{Lemme}

\begin{proof} %Borel
On va traiter le cas où $\overline{Y}$ est irréductible (le cas général se fait par la décomposition en irréductible\footnote{La noethérianité est importante, ici}. \\
Comme $Y$ est constructible, on a une décomposition $Y=\bigcup L_i$ où $L_i$ est une partie localement fermé de $X$. On a, alors, 
$\overline{Y}=\bigcup \overline{L_i}$. Par irréductibilité de $\overline{Y}$, il existe un $i$ tel que $\overline{Y}=\overline{L_i}$. $L_i$ est ouvert dans son adhérence car il est localement fermé et est donc un ouvert dense dans $\overline{Y}$.
\end{proof}
\begin{Lemme} \label{prod_dense}
Soit $G$ un groupe topologique. Soit $U$ un ouvert dense de $G$ et $V$ un ouvert non vide. Alors $UV=G$.
\end{Lemme}
\begin{proof}
Soit $g \in G$. Comme l'inversion $x \mapsto x^{-1}$ et les translations sont des homéomorphismes alors $gV^{-1}$ est un ouvert non vide de $G$. Comme $U$ est dense dans $G$ alors $U \cap gV^{-1}\neq \emptyset$. Ainsi, il existe $u \in U$ et $v \in V$ tel que $u=gv^{-1}$ i.e. $g=uv$.

\end{proof}
\begin{Prop}
Soit $G$ un groupe topologique et $H$ un sous-groupe de $G$. Si $H$ est une partie constructible de $G$ alors $H$ est fermée.

\end{Prop}

\begin{proof}
Soit $U$ un ouvert dense de $\overline{H}$ contenu dans $H$. Alors, par le lemme \ref{prod_dense}, $UU=\overline{H}$ mais comme $U \subset H$ alors $UU \subset HH=H$. Par conséquent, $H=\overline{H}$.

\end{proof}
\begin{Cor}\label{localement_ferme}
Soit $G$ un groupe topologique et $H$ un sous-groupe localement fermé de $G$. Alors $H$ est fermé dans $G$

\end{Cor}

\subsection{Groupes de Lie}

\begin{Def} \index{groupe de Lie}
Un groupe de Lie réel (resp. complexe)  est un groupe $(G,*)$ dont l'ensemble sous-jacent est muni d'une structure de variété lisse (resp. complexe) compatible avec les lois de groupes i.e. les applications $(x,y)  \in G \times G \mapsto x*y \in G$ et $x \in G \mapsto x^{-1} \in G$ sont lisses (resp. holomorphes). Autrement dit, un groupe de Lie réel (resp. complexe) est un groupe dans la catégorie des variétés différentielles (resp. variétés complexes).

\end{Def}

\begin{Def} \index{algebre de Lie@algèbre de Lie}
Soit $G$ un groupe de Lie (réel ou complexe). L'algèbre de Lie, notée $\gfrak$, associée à $G$ est l'espace vectoriel (réel ou complexe) des champs de vecteurs sur $G$ invariants à gauche i.e. les champs de vecteurs $X$ tels que, pour tout $g \in G$, $(L_g)_*X=X$ .
\end{Def}
L'espace vectoriel $\gfrak$ s'identifie à l'espace tangent de $G$ au neutre de la façon suivante : \\
A tout $x \in T_eG$, on associe le champ de vecteur $X : g \in G \mapsto d_e L_g(x)$ qui est invariant à gauche.
\begin{Def} \index{parallelisable @parallélisable}
Une variété (lisse ou complexe) est dite parallélisable si son fibré tangent est trivial.
\end{Def}

\begin{Thm}\label{trivialisable}
Un groupe de Lie (réel ou complexe) est parallélisable.

\end{Thm}

\begin{proof}
Soit $G$ un groupe de Lie (réel ou complexe). \\
On considère les applications \\
$\begin{array}{ccccc}
TG &\to& G \times \gfrak \\
(g,v) & \mapsto & (g,d_g L_{g^{-1}}(v))   \\
\end{array}$ et
$\begin{array}{ccccc}
 G \times \gfrak & \to & TG \\
 (x,v) & \mapsto & (x,d_e  L_g(v)) \\
\end{array}$ \\
Elles sont lisses (resp. holomorphes) et inverses l'une de l'autre. Ce sont donc des isomorphismes.
\end{proof}

De façon analogue au cas réel, on définit les sous-groupes à un paramètre comme les morphismes de groupes $\C \to G$. On a une bijection entre les sous-groupes à un paramètre et les éléments de l'algèbre de Lie $\gfrak$ : \\
A tout sous-groupe à un paramètre $\varphi : \C \to G$, on associe l'élément $d_0\varphi\left(\frac{d}{dz}\right) \in \gfrak$ et un élément $g \in \gfrak$, on associe l'unique morphisme $\varphi_g : \C \to G$ tel que $d_0\varphi\left(\frac{d}{dz}\right)=g$ que l'on définit par $t \in \C \mapsto \exp_{G_\R}(tg)$ où $\exp_{G_\R}$ est l'exponentielle entre les espaces réels sous-jacents (l'unicité découle de celle du cas réel). On définit l'exponentielle d'un élément $g \in \gfrak$ par :
\[\exp(g)=\varphi_g(1) \]

\begin{Thm} \label{exp}
Soit $G$ un groupe de Lie complexe et $\gfrak$ la $\C$-algèbre de Lie associée.
L'exponentielle réelle $\exp_\R : \gfrak_\R\to G_\R  $ entre les structures réelles des deux espaces induit une application holomorphe $\exp : \gfrak \to G$. 
\end{Thm}
\begin{proof}
Voir \cite{\Lee} Théorème 1.15 % Lee

 \end{proof}

\begin{Prop} \label{cccc}
Un groupe de Lie complexe connexe compact est commutatif.
\end{Prop}
\begin{proof}
Soit $G$ un groupe de Lie complexe connexe compact. \\
Soit $x \in X$. La conjugaison par $x$ dans $G$, $\sigma_x : y \in G \mapsto xyx^{-1}$ est un automorphisme de $G$. Sa différentielle est donc un isomorphisme. \\
Soit $\varphi : x \in G \mapsto d_e \sigma_x$. Cette application est holomorphe (car $\sigma_x$ (et donc ses dérivées partielles) le sont. Comme $X$ est compact alors $\varphi$ est localement constante et comme de plus, $X$ est connexe alors $\varphi$ est constante. \\
On en déduit : 
	\[ d_e \sigma_x=d_e \sigma_e=id_\gfrak
\]
On en déduit que, pour tout $u \in G$,  
	\[ \sigma_x(\exp(u))=\exp(d_e\sigma_x(u))=\exp(u)
\]
Ainsi, 
	\[\exp(\gfrak) \subset Z(G)
\]
De plus, par \ref{engendr}, 
	\[ G=\left\langle \exp(\gfrak) \right\rangle \subset Z(G) \subset G
\]
On en déduit que $Z(G)=G$ et donc que $G$ est commutatif.
\end{proof}

\begin{Thm} \label{Tore}
Soit $G$ un groupe de Lie complexe connexe compact. Alors l'application $\exp : \gfrak \to G$ est un morphisme surjectif de groupes dont le noyau est un réseau de $\gfrak$. On obtient un isomorphisme :
	\[ \gfrak/\Ker(\exp) \simeq G
\]

\end{Thm}

\begin{proof}
Soient $u,v \in \gfrak$ et $\psi : \C \to G$ définie par : \\
	\[ \forall t \in \C, \psi(t)=(\exp(tu)\exp(tv))=\varphi_u(t)\varphi_v(t)
\]
$\psi$ est une fonction holomorphe. Par la commutativité de $G$ donnée par \ref{cccc}, $\psi$ est un morphisme de groupes : 
	 
	 \begin{align*}
		 \forall t,t' \in \C, \psi(t+t')&=\varphi_u(t+t')\varphi_v(t+t')=\varphi_u(t)\varphi_u(t')\varphi_v(t)\varphi_v(t') \\
		&=\varphi_u(t)\varphi_v(t)\varphi_u(t')\varphi_v(t')=\psi(t)\psi(t')
	 \end{align*}

La différentielle de $\psi$ en 0 est donnée par : 
	\[ d_0 \psi=(u+v)dt
\]
Ainsi, par la correspondance précédant le théorème \ref{exp}, on en déduit que 
	\[ \psi=\varphi_{u+v}
\]
En évaluant en 1, on en déduit que : 
	\[ \exp(u+v)=\exp(u)\exp(v)
\]
i.e. $\exp$ est un morphisme de groupes. \\
On en déduit que $\exp(\gfrak)$ est un groupe et par \ref{engendr} et le fait que $\exp$ est un difféomorphisme local en 0 et que $G$ est connexe, 
	\[ \exp(\gfrak)=\left\langle \exp(\gfrak)\right\rangle=G
\]
Pour montrer que $\Ker(\exp)$ est discret, il suffit de le montrer au voisinage de 0 : 
Soit $U$ un voisinage ouvert de $0$ tel que $\exp_{|U}$ est injective (donnée par le théorème d'inversion locale). Alors,
$U \cap \Ker(\exp)=\{0\}$. \\
Soit $\widetilde{\exp} :\gfrak/\Ker(\exp) \to G$ l'application obtenue par passage au quotient. C'est un isomorphisme de groupes qui est un biholomorphisme local en 0 et donc en tout point par translation. 
\end{proof}


\subsection{$Aut_0 \subset Aut_\omega$} \label{demo_inclusion}
Le but de ce paragraphe est de montrer que, pour une variété de Kähler $(M,\omega)$, tous les éléments de la composantes de l'identité de $Aut(X)$ conserve la classe de $[\omega]$. On note par $Aut_\omega(X)$ le groupe des automorphismes de $X$ préservant la classe de $\omega$.

\begin{Thm}\label{autom_Lie}
Soit $M$ un variété complexe compacte. Son groupe d'automorphisme est un groupe de Lie complexe dont l'algèbre de Lie est constitué des champs de vecteurs holomorphes sur $M$.

\end{Thm}
\begin{proof}
Voir \cite{\BM} %Bochner Montgomery

\end{proof}

\begin{Lemme}\label{isotopie} % isotopie
Soient $M$ une variété lisse, $I$ un intervalle ouvert. Soit $\rho : M \times I \to M$ une isotopie lisse (i.e. pour tout $t \in I$, $\rho(\cdot,t)$ est un difféomorphisme). \\
Alors, pour toute forme $\omega$ et pour tout $t \in I$, 
	\[ \left[\frac{d}{ds} {\rho_s}_* \omega\right](t)=(\rho_t)_* \Lcal_{X_t} \omega
\]
où $(X_t)$ vérifie $\frac{d\rho_t}{dt} =X_t(\rho_t(\cdot))$ et $\Lcal$ est la dérivée de Lie.
\end{Lemme}
\begin{proof}
Voir théorème 36 p 175 \cite{\laf}
\end{proof}

\begin{Lemme}\label{derivee_forme}
Soit $s \in S \mapsto \omega_s$ une famille de formes différentielles fermées. Alors, 
	\[ \left[\frac{d \omega_s}{ds}\right]=\frac{d}{ds}\left[ \omega_s\right]
\]
où $\left[\omega\right]$ est la classe de cohomologie de $\omega$.
\end{Lemme}
\begin{proof}
Il faut seulement montrer que l'égalité ne dépend pas des classes choisies.\\
Soit $(\eta_s)$ une autre famille de formes différentielles fermées telle que, pour tout $s \in S$ $[\omega_s]=[\eta_s]$ i.e il existe une famille de formes différentielles exactes $(\gamma_s)$ telles que : 
	\[\forall s \in S,  \omega_s=\eta_s+d\gamma_s
\]
Ainsi, 
	\[\frac{d}{ds}\omega_s=\frac{d}{ds}\eta_s+\frac{d}{ds}d\gamma_s
\]
Or, par le lemme 44 de \cite{\laf} (page 179), $\frac{d}{ds}$ et $d$ commute et donc $\frac{d}{ds}d\gamma_s$ est nulle en cohomologie.
\end{proof}
\begin{Cor}\label{cohom_isotop}
Sous les hypothèses de \ref{isotopie}, si on prend une forme fermée $\omega$ et si on suppose $\rho_0=Id$, alors sa classe est préservée par l'isotopie i.e.
	\[ \forall t \in I, [\omega]=[{\rho_t}_* \omega]
\]
\end{Cor}
\begin{proof}
Par le lemme \ref{isotopie} , on a, pour tout $t \in T$, 
	\[ \left(\frac{d}{ds} {\rho_s}_* \omega\right)(t)=(\rho_t)_* \Lcal_{X_t} \omega
\]
Or, comme $\omega$ est fermé et que le pullback commute avec la différentielle, on a :
	\[ {\rho_t}_*\Lcal_{X_t} \omega={\rho_t}_*(di_{X_t} \omega+i_{X_t} d\omega)={\rho_t}_*di_{X_t}\omega=d({\rho_t}_*i_{X_t})\omega
\]
Ainsi, $\frac{d}{ds} {\rho_s}_*\omega$ est nulle en cohomologie et donc par le lemme \ref{derivee_forme}, ${\rho_s}_*\omega$ est constante et donc, pour tout $t\in T$, 
	\[[{\rho_t}_* \omega]=[{\rho_0}_*\omega]=[\omega]
\]

\end{proof}
\begin{Cor}\label{inclusion}
Soit $(M,\omega)$ un variété de Kähler compacte.
	\[Aut_0(M) \subset Aut_\omega(M)
\]


\end{Cor}

\begin{proof}
Par le théorème \ref{autom_Lie}, $Aut_0(M)$ est un groupe de Lie complexe dont l'algèbre de Lie est constitué des champs de vecteurs holomorphes sur $M$.
Par le théorème d'inversion locale, $\exp$ est un difféomorphisme autour de $0$, disons entre $U \subset \gfrak$ (que l'on supposera convexe) et $V \subset G$. Par le lemme \ref{engendr}, $V$ engendre $Aut_0(M)$. \\
Ainsi, un élément de $Aut_0(X)$ est de la forme 
	\[f=\exp(u_1)^{\varepsilon_1} \ldots \exp(u_n)^{\varepsilon_n}
\]
où $u_i \in U, \varepsilon_i \in \{\pm 1\}$. \\
On peut donc supposer que $f=\exp(u), u \in U$ et montrons que cette application préserve la classe de Kähler. On obtient le résultat voulu en itérant l'égalité obtenue. \\
L'application $f$ est isotope à l'identité grâce à l'application 
	\[\rho : (t,x) \in ]-\varepsilon,1+\varepsilon[ \mapsto \exp(tu)(x)
\]
 pour $\varepsilon$ bien choisi ( $\varepsilon>0$ est choisi de tel sorte que $-\varepsilon u $ et $(1+\varepsilon)u$ soient dans $U$).
Les $\rho(t,\cdot)$ sont toujours des difféomorphismes grâce au fait que $U$ est convexe et donc $]-\varepsilon,1+\varepsilon[u \subset U$. Par conséquent, on obtient notre résultat grâce au corollaire \ref{cohom_isotop}, 

\end{proof}
