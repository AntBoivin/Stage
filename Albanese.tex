Soit $X$ une variété de Kähler connexe de dimension complexe $n$. \\
Par la décomposition de Hodge (\ref{Hodge}), \\
	\[ H^{2n-1}(X,\C)=H^{n,n-1}(X,\C) \oplus \overline{H^{n,n-1}(X,\C)}
\]
On en déduit que 
	\[ H^{2n-1}(X,\R) \cap H^{n,n-1}(X,\C)=\emptyset
\]
 car, dans le cas contraire, un point de l'intersection serait dans l'intersection $H^{n,n-1}(X) \cap \overline{H^{n,n-1}(X)}$ (car réel) qui est vide.
L'application 
	\[H^{2n-1}(X,\R) \hookrightarrow H^{2n-1}(X,\C) \twoheadrightarrow H^{2n-1}(X)/H^{n,n-1}(X)
\]
 est injective et donc est un isomorphisme (car ils ont la même dimension).
Par la dualité de Poincaré(\ref{lem_poinc}), on a $H^{2n-1}(X,\R)=H_1(X,\R)$ et par celle de Kodaira-Serre (\cite{\pag} p 102), 
	\[ \overline{H^{n,n-1}(X,\C)}=H^{n-1,n}(X) \simeq H^{1,0}(X)^*=H^0(X,\Omega_X)^*
\]
Ainsi, $H_1(X,\Z)^{libre}$ est un réseau de $H^0(X,\Omega_X)^*$.

\begin{Def} \index{variete d'Albanese@variété d'Albanese}
Le quotient $H^0(X,\Omega_X)^*/H_1(X,\Z)^{libre}$ est appelé variété (ou tore) d'Albanese, que l'on note $Alb(X)$.
\end{Def}

On associe à cette variété une application $alb_X : X \mapsto Alb(X)$ définie, en fixant un point $x_0 \in X$, par : 
	\[ \forall x \in X, alb_X(x) : [\omega] \mapsto \int_{x_0}^x \omega
\]
Cette application ne dépend pas du point base choisie grâce au quotient par $H_i(X,\Z)^{libre}$. \\

\begin{Prop} \label{engend_alb}
L'application $alb_X : X \to Alb(X)$ est holomorphe et son image engendre $Alb(X)$ en tant que groupe.

\end{Prop}

\begin{proof}
La première partie est un cas particulier d'un résultat de Griffiths dont la preuve peut être trouvé dans \cite{\voisin} (Théorème 12.4 p 274) et la deuxième se trouve au même endroit (lemme 12.11 p 278).

\end{proof}

\begin{Lemme}\label{tore_alb}
Si $\Tcal$ est un tore complexe alors $Alb(\Tcal)=\Tcal$ et $alb_\Tcal$ est une translation.
\end{Lemme}

\begin{proof}
Par définition, on peut écrire $\Tcal$ sous la forme $\C^n/\Lambda$ où $\Lambda$ est un réseau de $\C^n$. \\
Comme $T$ est un groupe de Lie complexe alors il est parallélisable (voir \ref{trivialisable}) i.e 
	\[T\Tcal=\Tcal \times \C^n 
\]
Une forme différentielle est une application holomorphe 
	\[\omega : \Tcal \to T^*\Tcal=\Tcal \times (\C^n)^*
\]
 tel que $\omega(x) \in T_x \Tcal$. Dans notre cas, une forme différentielle se résume à une application holomorphe $\Tcal \to (\C^n)^*$ et donc comme $\Tcal$ est compacte alors elles sont constantes (et donc $alb_\Tcal$ est une translation). Ainsi, 
	\[ H^0(\Tcal,\Omega_\Tcal)=(\C^n)^*
\]
Ensuite, comme $\C^n$ est simplement connexe et que $\Tcal$ est une variété alors $\C^n$ est le revêtement universel de $\Tcal$ et donc, pour tout $x \in \Tcal$, 
	\[\pi_1(\Tcal,x)=\Lambda
\]
Et donc, 
	\[H^1(\Tcal,\Z)=\pi_1(\Tcal,x)^{ab}=\Lambda^{ab}=\Lambda
\]
On en déduit que :
	\[ \Tcal=\C^n/\Lambda=H^0(\Tcal,\Omega_\Tcal)^*/H_1(\Tcal,\Z)^{libre}=Alb(\Tcal)
\]
\end{proof}

\begin{Lemme} \label{holom_tore}
Si $f : \Tcal_1 \to \Tcal_2$ est une application holomorphe entre deux tores complexes alors $f$ est un morphisme de groupes suivi d'une translation.

\end{Lemme}

\begin{proof}
Quitte à tranlater, on peut supposer $f(0)=0$. \\
On peut écrire les deux tores sous la forme $\Tcal_1=\C^n/\Lambda_1$ et $\Tcal_1=\C^m/\Lambda_2$. Ces deux tores sont parallélisables et donc $T\Tcal_1=\Tcal_1 \times \C^n$ et $T\Tcal_2=\Tcal_2 \times \C^m$. L'application $df$ est donc une application $\Tcal_1 \times \C^n \to \Tcal_2 \times \C^m$. Si on fixe $v \in \C^m$, on peut définir une application holomorphe $x \in T_1 \mapsto d_xf(v) \in \C^m$. Celle-ci est constante car $\Tcal_1$ est compacte. On en déduit que $d_xf(v)$ ne dépend pas de $x$ et donc $df$ est uniquement déterminé par une applcation $\C$-linéaire $\C^n \to \C^m$ (que l'on notera $A$). \\
Soit $\overline{f} : \C^n \to \C^m$ l'application qui relève $\pi_2 \circ f : \Tcal_1 \to \C^m$ et telle que $\overline{f}(0)=0$. Il suffit donc de montrer que $\overline{f}=A$. Montrons le au voisinage de 0. \\
La différence de $f$ et celle de $\overline{f}$ coïncide (car la projection est un difféomorphisme local). Soit $z$ dans un voisinage de $0$et $\gamma$ un chemin de $0$ et $z$. Alors, 
	\[\overline{f}(z)=\int_0^1 d_{\gamma(t)}\overline{f}(\gamma'(t))dt=\int_0^1 A(\gamma'(t))dt=A(\gamma(1))-A(\gamma(0))=A(z)
\]
Par le prolongement analytique, on obtient $\overline{f}=A$
\end{proof}

\begin{Thm}
Soit $X$ une variété de Kähler. \\
$(Alb(X),alb_X)$ vérifie la propriété universelle suivante : 
Pour toute application holomorphe $f : X \to T$ où $T$ est un tore complexe, il existe une unique application holomorphe $g : Alb(X)\to T$ telle que le diagramme suivant commute
 \xymatrix{
    X \ar[r]^f \ar[d]_{alb_X}  & T \  \\
    Alb(X) \ar[ru]_g & 
  }

\end{Thm}

\begin{proof}
Existence : \\
Par naturalité, il existe une application $f_* : H_0(X,\Omega_X)^* \to H_0(T,\Omega_T)^*$ qui se restreint en une application $f_* : H_1(X,\Z) \to H_1(T,\Z)$. Par passage au quotient, on obtient une application $f_* : Alb(X)\to Alb(T)$. On remarque que cette application vérifie l'égalité $alb_T \circ f=f_* \circ alb_X$ en envoyant le point base de $X$ sur celui de $T$ (i.e est naturelle). Comme $alb_T$ est une translation (par \ref{tore_alb}) donc est inversible. On en déduit que $f=alb_T^{-1} \circ f_* \circ alb_X$. \\
Unicité : \\
Soit $g_1,g_2$ deux solutions du problème. Posons $g:=g_1-g_2$. \\
L'application $g$ est une application holomorphe  $Alb(X) \to T$ nulle sur $alb_X(X)$ et en particulier en $0$. Par \ref{holom_tore}, c'est une morphisme de groupes. Comme $alb_X(X)$ engendre $Alb(X)$ par \ref{engend_alb} alors $g$ est nulle sur tout $Alb(X)$.
\end{proof}

\begin{Thm} \label{Aut_0_Alb}
Soit $X$ une variété de Kähler compacte. Alors,
    \[ Aut_0(Alb(X)) \simeq Alb(X)
\]
\end{Thm}

\begin{proof}
Par la proposition \ref{holom_tore}, un automorphisme de $Alb(X)$ est la composée d'un automorphisme de groupes suivi d'une translation. \\
Comme $Aut_0(X)$ est un groupe de Lie connexe alors en particulier, il est connexe par arcs. Alors, les automorphismes de $Aut_0(X)$ sont ceux atteignables par un arc à partir de l'identité.
Commençons au voisinage de l'identité : \\
Soit $(f_n)$ une suite de $Aut_0(Alb(X))$ convergeant vers l'identité. On peut décomposer $f_n$ sous la forme $\varphi_n+\lambda$ où $\varphi_n$ est un automorphisme de groupes et $\lambda_n$ une constante. \\
Soit $\varepsilon>0$. Comme $(f_n)$ converge vers l'identité et que $Aut(X)$ est muni de la topologie de la convergence uniforme alors il existe $N \in \N$ tel que pour tout $n \geq N$,
    \[ \|Id-f_n\|_\infty<\varepsilon
\]
i.e pour tout $x \in X$,
    \[|x-f_n(x)|<\varepsilon
\]
En particulier, pour $x=0$,
\[|\lambda_n|<\varepsilon
\]
Supposons que $\varphi_n \neq Id$ i.e il existe $x \in X$ tel que $\varphi_n(x) \neq x$. Pour $\varepsilon$ suffisamment petit, il existe $m \in \N$ tel que
    \[m|\varphi_n(x)-x|=|\varphi_n(mx)-mx|>2\varepsilon
\]
(car $\varphi_n$ est un morphisme de groupes). Ainsi, $\|Id-\varphi_n\|_\infty>2\varepsilon$. \\
On obtient ainsi une contradiction car :
    \[ \|Id-\varphi_n\|_\infty \leq \|Id-f_n\|_\infty+|\lambda_n|\leq 2 \varepsilon
\]
On en déduit que $\varphi_n=Id$. \\
Soient $f \in Aut_0(Alb(X))$ et un arc $\gamma : [0,1] \to Aut_0(X)$ reliant $Id$ et $f$. En recouvrant $\gamma([0,1])$ par un nombre (fini) d'ouverts suffisamment petit, on en déduit, grâce à la discussion précédente que, pour tout $t \in [0,1]$, $\gamma(t)$ est une translation. \\
L'isomorphisme $Aut_0(Alb(X)) \to Alb(X)$ est donnée par $f \mapsto f(0)$.
\end{proof}
